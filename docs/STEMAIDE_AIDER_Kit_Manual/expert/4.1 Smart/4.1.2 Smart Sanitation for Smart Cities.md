```python

# Smart Sanitation for Smart Cities - Project 352

# Three-layer IoT Architecture

from machine import Pin, ADC
import time

# Layer 1: Facility Node Sensors
usage_sensor = Pin(14, Pin.IN)
odor_sensor = ADC(Pin(26))

def read_facility_data():
    """Layer 1: Read usage and odor sensors"""

    usage_count = usage_sensor.value()

    # Read odor level
    odor_raw = odor_sensor.read_u16()
    odor_level = int(((65535 - odor_raw) / 65535) * 100)

    return {
        "usage_detected": usage_count,
        "odor_level": odor_level,
        "timestamp": time.time()
    }

def assess_sanitation(data):
    """Layer 2: Sanitation condition logic"""

    odor = data["odor_level"]

    if odor < 30:
        return "SERVICE_REQUIRED", "CRITICAL"
    elif odor < 40:
        return "SERVICE_REQUIRED", "MODERATE"
    elif odor < 60:
        return "MONITOR", "OK"
    else:
        return "GOOD", "OK"

def generate_facility_status(facility_id, data, condition):
    """Layer 3: Dashboard status"""

    status, alert = condition

    return {
        "facility_id": facility_id,
        "status": status,
        "alert_level": alert,
        "last_service": time.time() - 3600,
        "next_action":
            "SCHEDULE_SERVICE"
            if status == "SERVICE_REQUIRED"
            else "MONITOR"
    }

# Main loop
def start_sanitation_system():

    facility_id = "FACILITY_001"

    while True:

        # Layer 1: Collect
        data = read_facility_data()

        # Layer 2: Process
        condition = assess_sanitation(data)

        # Layer 3: Output
        status = generate_facility_status(
            facility_id,
            data,
            condition
        )

        print(f"Facility: {facility_id}")
        print(f" Usage: {data['usage_detected']}")
        print(f" Odor: {data['odor_level']}%")
        print(f" Status: {status['status']}")
        print(f" Alert: {status['alert_level']}")
        print("-" * 40)

        time.sleep(10)

start_sanitation_system()
```

## Troubleshooting

| Problem                        | Possible Cause                                         | Solution                                                                  |
| ------------------------------ | ------------------------------------------------------ | ------------------------------------------------------------------------- |
| No readings appear in Thonny   | Incorrect wiring, wrong pins, or missing library       | Recheck the circuit, confirm pin numbers, and test each module separately |
| Import error appears           | Library file is missing or saved in the wrong location | Check the Pico file structure and save libraries exactly as instructed    |
| Actuator does not respond      | Relay/driver wiring or external power issue            | Verify driver input, external power, and shared ground where required     |
| Results are unstable           | Sensor noise, loose wiring, or poor calibration        | Secure the wiring and add calibration or averaging in code                |
| Prototype works only sometimes | System is too complex to test all at once              | Return to subsystem testing and integrate one part at a time              |
