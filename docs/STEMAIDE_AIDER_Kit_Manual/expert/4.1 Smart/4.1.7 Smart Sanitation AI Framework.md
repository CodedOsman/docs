### Starter Code

```python

# Smart Sanitation AI Framework - Project 357

# AI-Enhanced Multi-Parameter Monitoring

from machine import Pin, ADC
import time

usage = Pin(14, Pin.IN)
odor = ADC(Pin(26))
waste = ADC(Pin(27))

# Layer 1: Facility Edge Node
def read_sanitation_data():
    usage_count = usage.value()
    odor_raw = odor.read_u16()
    waste_raw = waste.read_u16()

    odor_level = int(((65535 - odor_raw) / 65535) * 100)
    waste_level = int(((65535 - waste_raw) / 65535) * 100)

    return {
        "usage": usage_count,
        "odor": odor_level,
        "waste": waste_level
    }

# Layer 2: AI Index with Weights
def calculate_ai_index(data):

    # AI-inspired weighted scoring
    odor_w = 0.5
    waste_w = 0.3
    usage_w = 0.2

    odor_score = data["odor"]
    waste_score = data["waste"]
    usage_score = 100 if data["usage"] else 20

    ai_score = (
        (odor_score * odor_w)
        + (waste_score * waste_w)
        + (usage_score * usage_w)
    )

    if ai_score >= 75:
        return "HEALTHY", ai_score
    elif ai_score >= 50:
        return "MONITOR", ai_score
    elif ai_score >= 25:
        return "WARNING", ai_score
    else:
        return "CRITICAL", ai_score

# Layer 3: Smart Nation Layer
def generate_national_status(status, score):

    actions = {
        "HEALTHY": [
            "Continue monitoring",
            "Routine maintenance"
        ],
        "MONITOR": [
            "Schedule inspection",
            "Increase frequency"
        ],
        "WARNING": [
            "Urgent cleaning required",
            "Alert supervisor"
        ],
        "CRITICAL": [
            "Immediate action",
            "Block facility"
        ]
    }

    return actions.get(status, [])

def run_demo():

    scenarios = [
        (1, 80, 85),
        (0, 50, 40),
        (1, 20, 15)
    ]

    for usage_val, odor_level, waste_level in scenarios:

        data = {
            "usage": usage_val,
            "odor": odor_level,
            "waste": waste_level
        }

        status, score = calculate_ai_index(data)
        actions = generate_national_status(status, score)

        print(
            f"\nUsage: {usage_val}, "
            f"Odor: {odor_level}%, "
            f"Waste: {waste_level}%"
        )

        print(f"AI Score: {score:.1f} - {status}")

        for action in actions:
            print(f"  - {action}")

run_demo()
```

## Troubleshooting

| Problem                        | Possible Cause                                         | Solution                                                                  |
| ------------------------------ | ------------------------------------------------------ | ------------------------------------------------------------------------- |
| No readings appear in Thonny   | Incorrect wiring, wrong pins, or missing library       | Recheck the circuit, confirm pin numbers, and test each module separately |
| Import error appears           | Library file is missing or saved in the wrong location | Check the Pico file structure and save libraries exactly as instructed    |
| Actuator does not respond      | Relay/driver wiring or external power issue            | Verify driver input, external power, and shared ground where required     |
| Results are unstable           | Sensor noise, loose wiring, or poor calibration        | Secure the wiring and add calibration or averaging in code                |
| Prototype works only sometimes | System is too complex to test all at once              | Return to subsystem testing and integrate one part at a time              |
