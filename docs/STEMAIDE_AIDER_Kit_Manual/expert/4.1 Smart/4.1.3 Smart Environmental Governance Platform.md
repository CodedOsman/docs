```python
# Smart Environmental Governance Platform - Project 354

from machine import Pin, ADC
import time

air_sensor = ADC(Pin(26))
water_sensor = ADC(Pin(27))

# Regulatory thresholds (simplified WHO/EPA standards)
COMPLIANCE_THRESHOLDS = {
    "air_quality_min": 60,
    "water_quality_min": 70
}

def read_environmental_data():
    """Layer 1: Compliance Monitoring Node"""
    air_raw = air_sensor.read_u16()
    water_raw = water_sensor.read_u16()

    air_quality = int(((65535 - air_raw) / 65535) * 100)
    water_quality = int(((65535 - water_raw) / 65535) * 100)

    return {
        "air_quality": air_quality,
        "water_quality": water_quality,
        "timestamp": time.time()
    }

def determine_compliance(data):
    """Layer 2: Compliance Logic"""
    air = data["air_quality"]
    water = data["water_quality"]

    air_compliant = air >= COMPLIANCE_THRESHOLDS["air_quality_min"]
    water_compliant = water >= COMPLIANCE_THRESHOLDS["water_quality_min"]

    if air_compliant and water_compliant:
        return "COMPLIANT", "PASS"
    elif not air_compliant and not water_compliant:
        return "NON-COMPLIANT", "FAIL_BOTH"
    elif not air_compliant:
        return "NON-COMPLIANT", "FAIL_AIR"
    else:
        return "NON-COMPLIANT", "FAIL_WATER"

def generate_governance_report(site_id, data, compliance):
    """Layer 3: Governance Dashboard"""

    status, detail = compliance

    report = {
        "site_id": site_id,
        "status": status,
        "detail": detail,
        "air_quality": data["air_quality"],
        "water_quality": data["water_quality"],
        "thresholds": COMPLIANCE_THRESHOLDS,
        "audit_timestamp": time.time(),
        "required_actions": [] if status == "COMPLIANT" else [
            "Issue compliance notice" if "AIR" in detail else "",
            "Issue water quality warning" if "WATER" in detail else "",
            "Schedule inspection"
        ]
    }

    return report

def start_governance_system():
    site_id = "INDUSTRIAL_SITE_001"

    while True:
        data = read_environmental_data()
        compliance = determine_compliance(data)
        report = generate_governance_report(site_id, data, compliance)

        print(f"\n{'='*50}")
        print("ENVIRONMENTAL GOVERNANCE REPORT")
        print(f"{'='*50}")
        print(f"Site: {report['site_id']}")
        print(f"Status: {report['status']}")
        print(f"Detail: {report['detail']}")

        print("\nReadings:")
        print(f" Air Quality: {report['air_quality']}%")
        print(f" Water Quality: {report['water_quality']}%")

        print("\nRequired Actions:")
        for action in report['required_actions']:
            if action:
                print(f" - {action}")

        print(f"{'='*50}")

        time.sleep(20)

start_governance_system()
```

## Troubleshooting

| Problem                        | Possible Cause                   | Solution                                      |
| ------------------------------ | -------------------------------- | --------------------------------------------- |
| No readings appear in Thonny   | Incorrect wiring or wrong pins   | Verify all wiring and sensor connections      |
| Import error appears           | Missing file or library          | Check file locations and library installation |
| Actuator does not respond      | Driver or power issue            | Verify external power and shared grounds      |
| Results are unstable           | Sensor noise or poor calibration | Improve calibration and average readings      |
| Prototype works only sometimes | System complexity                | Test and integrate one subsystem at a time    |
