## Overview

Build a relay scheduler that uses Wi-Fi time sync, local buttons, an OLED display, and a browser page.

This project demonstrates using internet time to trigger relay actions at chosen times of day.

The final result should show the current time on the OLED, allow manual control from buttons or browser, and switch the relay automatically when a scheduled time is reached.

## Full Project Code

Upload and run this code after the individual tests work correctly.

```python
import network
import ntptime
import socket
import time
from machine import I2C, Pin
import sh1106

# =========================

# Wi-Fi Credentials

# =========================
SSID = 'YOUR_WIFI_NAME'
PASSWORD = 'YOUR_WIFI_PASSWORD'

# =========================

# Relay Setup

# Active LOW relay:

# 0 = ON

# 1 = OFF

# =========================
relay = Pin(0, Pin.OUT)
relay.value(1)

# =========================

# Single Toggle Button

# =========================
btn_toggle = Pin(15, Pin.IN, Pin.PULL_UP)

# =========================

# OLED Setup

# =========================
i2c = I2C(0, sda=Pin(8), scl=Pin(9), freq=400000)
oled = sh1106.SH1106_I2C(128, 64, i2c)

# =========================

# Timezone Offset

# Ghana = 0

# Example:

# Nigeria = 1

# India = 5

# =========================
TIMEZONE_OFFSET = 0

# =========================

# Relay Schedule

# Format:

# (hour, minute, action)

# action:

# 1 = ON

# 0 = OFF

# =========================
schedule = [
    (7, 0, 1),
    (8, 30, 0),
    (18, 0, 1),
    (22, 0, 0),
]

# =========================

# Time Functions

# =========================
def local_hour_minute():
    now = time.localtime()
    hour = (now[3] + TIMEZONE_OFFSET) % 24
    minute = now[4]
    return hour, minute

# =========================

# Relay Control

# =========================
def relay_on():
    relay.value(0)

def relay_off():
    relay.value(1)

def relay_toggle():
    if relay.value() == 1:
        relay_on()
    else:
        relay_off()

def relay_status_text():
    return 'ON' if relay.value() == 0 else 'OFF'

# =========================

# OLED Display

# =========================
def update_oled():
    hour, minute = local_hour_minute()

    oled.fill(0)

    oled.text('Relay Scheduler', 0, 0, 1)
    oled.text('{:02d}:{:02d}'.format(hour, minute), 32, 18, 1)
    oled.text('Relay: ' + relay_status_text(), 10, 42, 1)

    oled.show()

# =========================

# Web Page

# =========================
def web_page():
    hour, minute = local_hour_minute()

    return '''<!DOCTYPE html>
<html>
<head>
    <meta name='viewport' content='width=device-width, initial-scale=1'>
    <meta http-equiv='refresh' content='5'>
    <title>Relay Scheduler</title>
</head>

<body style='font-family:Arial;text-align:center;padding:30px;background:#f2f2f2;'>

    <h1>Wi-Fi Relay Scheduler</h1>

    <h2>{:02d}:{:02d}</h2>

    <p><b>Relay State:</b> {}</p>

    <p>
        <a href='/on'>
            <button style='padding:15px 30px;font-size:18px;'>ON</button>
        </a>

        <a href='/off'>
            <button style='padding:15px 30px;font-size:18px;'>OFF</button>
        </a>
    </p>

    <h3>Schedule</h3>

    <p>07:00 ON</p>
    <p>08:30 OFF</p>
    <p>18:00 ON</p>
    <p>22:00 OFF</p>

</body>
</html>
'''.format(hour, minute, relay_status_text())

# =========================

# Connect to Wi-Fi

# =========================
wlan = network.WLAN(network.STA_IF)
wlan.active(True)
wlan.connect(SSID, PASSWORD)

print('Connecting to Wi-Fi...')

for _ in range(15):
    if wlan.isconnected():
        break

    time.sleep(1)

if not wlan.isconnected():
    raise RuntimeError('Wi-Fi connection failed')

print('Wi-Fi Connected')

# =========================

# Sync Time from Internet

# =========================
try:
    ntptime.settime()
    print('Time synced with NTP')

except Exception as error:
    print('NTP sync failed:', error)

# =========================

# Show IP Address

# =========================
ip_address = wlan.ifconfig()[0]

print('Open this in browser:')
print('http://{}'.format(ip_address))

# =========================

# Start Web Server

# =========================
address = socket.getaddrinfo('0.0.0.0', 80)[0][-1]

server = socket.socket()

server.bind(address)
server.listen(1)

server.settimeout(0.2)

print('Web server started')

# =========================

# Schedule Tracking

# =========================
last_checked_minute = None

# =========================

# Main Loop

# =========================
while True:

    # =====================
    # Single Button Toggle
    # =====================
    if btn_toggle.value() == 0:

        time.sleep_ms(50)

        if btn_toggle.value() == 0:

            relay_toggle()

            print('Button Toggle:', relay_status_text())

            # Wait until released
            while btn_toggle.value() == 0:
                pass

            time.sleep_ms(50)

    # =====================
    # Schedule Control
    # =====================
    current_hour, current_minute = local_hour_minute()

    minute_signature = (current_hour, current_minute)

    if minute_signature != last_checked_minute:

        for schedule_hour, schedule_minute, action in schedule:

            if (current_hour == schedule_hour and
                    current_minute == schedule_minute):

                if action == 1:
                    relay_on()
                else:
                    relay_off()

                print('Schedule Action:',
                      'ON' if action == 1 else 'OFF')

        last_checked_minute = minute_signature

    # =====================
    # Update OLED
    # =====================
    update_oled()

    # =====================
    # Handle Web Requests
    # =====================
    try:
        client, client_address = server.accept()

    except OSError:
        continue

    request = client.recv(1024).decode()

    if 'GET /on' in request:
        relay_on()

    elif 'GET /off' in request:
        relay_off()

    response = web_page()

    client.send(
        'HTTP/1.1 200 OK\r\n'
        'Content-Type: text/html\r\n'
        'Connection: close\r\n\r\n'.encode()
    )

    client.sendall(response.encode())

    client.close()
```

## How the Code Works

| Code Section    | What It Does                                         | Why It Matters                                   |
| --------------- | ---------------------------------------------------- | ------------------------------------------------ |
| Schedule list   | Stores times and relay actions                       | This is the core of the scheduler project        |
| NTP sync        | Tries to get the current time from the internet      | The schedule needs a real time reference         |
| Manual controls | Allow the relay to be switched by buttons or browser | This gives the user control outside the schedule |
| OLED update     | Shows the current time and relay state               | The scheduler becomes easier to observe locally  |

## Troubleshooting

| Problem                   | Possible Cause                                       | Solution                                                                     |
| ------------------------- | ---------------------------------------------------- | ---------------------------------------------------------------------------- |
| Time is wrong             | Timezone offset is not set for your location         | Adjust TIMEZONE_OFFSET in the code                                           |
| NTP sync fails            | The board has Wi-Fi but cannot reach the time server | Check internet access on the network and try again                           |
| OLED is blank             | Wrong I2C pins or missing sh1106.py                  | Use GPIO 8/9 and confirm sh1106.py is on the Pico                            |
| Schedule does not trigger | Current time or schedule list does not match         | Print the current hour and minute and compare them with the schedule entries |
